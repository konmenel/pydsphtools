def get_dsph_binpath(cpu: bool, rootdir: str) -> str:
    if not os.path.exists(rootdir):
        raise FileNotFoundError(errno.ENOENT, os.strerror(errno.ENOENT), rootdir)

    plat = platform.system()
    dirbin = Path(rootdir) / "bin"
    if plat == "Linux":
        dirbin /= "linux"
        file_prefix = "_linux64"
    elif plat == "Windows":
        dirbin /= "windows"
        file_prefix = "_win64.exe"

    files = (f for f in os.listdir(dirbin) if os.path.isfile(os.path.join(dirbin, f)))

    regex = r"DualSPHysics\d+\.\d+"
    if cpu:
        regex += "CPU"
    regex += file_prefix
    regex = re.compile(regex)

    for file in files:
        if regex.match(file):
            return os.path.join(dirbin, file)

def get_tool_binpath(toolname: str, rootdir: str) -> str:
    if not os.path.exists(rootdir):
        raise FileNotFoundError(errno.ENOENT, os.strerror(errno.ENOENT), rootdir)

    plat = platform.system()
    dirbin = Path(rootdir) / "bin"
    if plat == "Linux":
        dirbin /= "linux"
        file_prefix = "_linux64"
    elif plat == "Windows":
        dirbin /= "windows"
        file_prefix = "_win64.exe"

    files = (f for f in os.listdir(dirbin) if os.path.isfile(os.path.join(dirbin, f)))

    regex = toolname
    regex += file_prefix
    regex = re.compile(regex, re.IGNORECASE)

    for file in files:
        if regex.match(file):
            return os.path.join(dirbin, file)

class Runner:
    parser: argparse.ArgumentParser
    _args: argparse.Namespace

    _cpu: bool
    _dsph_root: str
    _run_gencase: bool
    _run_dsph: bool

    _gencase_options = List[str]
    _dsph_options = List[str]
    _gencase_binpath: str
    _dsph_binpath: str

    _exitcode: int

    def __init__(
        self,
        cpu: bool = True,
        run_gencase: bool = True,
        run_dsph: bool = True,
        rootdir: str = None,
    ) -> None:
        if rootdir is None and "DUALSPH_HOME" not in os.environ:
            message = (
                "Environment variable 'DUALSPH_HOME' is missing and no path "
                "for the parent directory of SPH was provided. Please either "
                "define the environment variable 'DUALSPH_HOME' or pass the "
                "root directory of DualSPHysics through the variable `binpath`."
            )
            raise MissingEnvironmentVariable("DUALSPH_HOME", message)

        if rootdir is None:
            rootdir = os.environ["DUALSPH_HOME"]

        self._cpu = cpu
        self.parser = argparse.ArgumentParser()
        self._args = None
        self._dsph_root = rootdir
        self._dsph_binpath = get_dsph_binpath(cpu, rootdir)
        self._gencase_binpath = get_tool_binpath("gencase", rootdir)
        self._run_gencase = run_gencase
        self._run_dsph = run_dsph
        self._gencase_options = []
        self._dsph_options = []
        self._exitcode = 0

        if not cpu:
            self._dsph_options.append("-gpu")

        self._fix_ld_library_path()

    @property
    def cpu(self) -> bool:
        return self._cpu

    @property
    def exitcode(self) -> int:
        return self._exitcode

    @property
    def args(self) -> argparse.Namespace:
        return self._args

    def enable_gpu(self) -> None:
        if self._cpu:
            self._cpu = False
            self._dsph_binpath = get_dsph_binpath(False, self._dsph_root)
            self._dsph_options.append("-gpu")

    def enable_cpu(self) -> None:
        if not self._cpu:
            self._cpu = True
            self._dsph_binpath = get_dsph_binpath(True, self._dsph_root)
            for opt in self._dsph_options:
                if opt.startswith("-gpu"):
                    self._dsph_options.remove(opt)

    def enable_gencase_run(self) -> None:
        self._run_gencase = True

    def enable_dsph_run(self) -> None:
        self._run_dsph = True

    def disable_gencase_run(self) -> None:
        self._run_gencase = False

    def disable_dsph_run(self) -> None:
        self._run_dsph = False

    def is_gencase_active(self) -> bool:
        return self._run_gencase

    def is_dsph_active(self) -> bool:
        return self._run_dsph

    def add_gencase_option(self, option: str) -> None:
        self._gencase_options.append(option)

    def add_dsph_option(self, option: str) -> None:
        self._dsph_options.append(option)

    def add_gencase_options(self, options: Iterable[str]) -> None:
        self._gencase_options.extend(options)

    def add_dsph_options(self, options: Iterable[str]) -> None:
        self._dsph_options.extend(options)

    def add_cl_argument(self, *args, **kwargs):
        self.parser.add_argument(*args, **kwargs)

    def parse(self) -> None:
        if self._args is None:
            self._args = self.parser.parse_args()

    def run(self, fail: bool = True) -> int:
        self.parse()
        if self.is_gencase_active():
            ret = subprocess.run(
                [self._gencase_binpath, *self._gencase_options], check=fail
            )
            self._exitcode = ret.returncode

        if self._exitcode != 0:
            print("[INFO] Runner: GenCase exited with non-zero exit code.")
            return self.exitcode

        if self.is_dsph_active():
            ret = subprocess.run([self._dsph_binpath, *self._dsph_options], check=fail)
        self._exitcode = ret.returncode

        if self._exitcode != 0:
            print("[INFO] Runner: DualSPHysics exited with non-zero exit code.")

        return self._exitcode

    def _fix_ld_library_path(self) -> None:
        ld_library_path = os.environ["LD_LIBRARY_PATH"]
        plat = platform.system()
        dirbin = Path(self._dsph_root) / "bin"
        if plat == "Linux":
            dirbin /= "linux"
        elif plat == "Windows":
            dirbin /= "windows"

        if ld_library_path:
            os.environ["LD_LIBRARY_PATH"] = ld_library_path + ":" + str(dirbin)
        else:
            os.environ["LD_LIBRARY_PATH"] = str(dirbin)
